;******************************************************************************

; PROGRAM FOR CONTROLLING THE DISPLAY MODULE WITH lC7981 SANYO CONTROLLER USING 89C52

 & LCD CONTROLLER

; DATE : 06 AUG 1999.

;******************************************************************************

 

; The following program first of all sets the LCD Module in the TEXT Mode.

; It starts with the message "STARTING TEXT MODE..."

; Then it generates all the characters available in the INTERNAL CG RAM.

; After having generated the said characters, the display mode is set to

; GRAPHIC Mode when the following message has appeared

; "WAIT... INITIALIZING GRAPHIC MODE"

; In the GRAPHIC Mode the display is Blanked first, alternate lines are

; shown for next two cycles, the screen is then Blacked out.

; The display is Blanked again to show the message "TESTING OVER..."

 

;******************************************************************************

 

;     THE PIN CONNECTIONS FOR THE LCD CONTROLLER AND 89C52 ARE AS GIVEN BELOW

;     ---------------------------------------------------------------

 

;     PIN NO. OF  20 PIN CONNECTOR         89C52 PIN CONNECTIONS

;     AND CONNECTION TO LCD CONTROLLER  <--->as available in the Hardware

;      ============================        ============================

 

;     1       =      Vss               GND

;     2       =       VDD               Vcc

;     3      =      Vo                -

;     4      =      RS                P2.0

;     5      =      R/W!              WR! ( P3.6 )

;     6      =      E                 The O/P of a NAND Gate with

;                                   RD! & WR! of 89C52 as I/Ps of

;                                   the NAND Gate. ( RD! = P3.7 )

;                                   This signal is meant for latch.

;     7 - 14      =      DB0 - DB7         P0.0 - P0.7

;     15      =      CS!               P2.1

;     16      =      RES!              P3.1

;     17      =      Vee               -

;     18      =      NC                -

;     19      =      NC                -

;     20      =      NC                -

 

;******************************************************************************

 

;******************************

; SPACE TO DEFINE THE CONSTANTS

;******************************

 

R3_2        EQU      1AH         ; CONST FOR DELAY

DATA_REG      EQU      1BH         ; CONST FOR DELAY

 

; RESPECTIVE CONSTANTS FOR TEXT MODE

 

MODE        EQU      3CH

PITCH_DT      EQU      75H

NUMCHR            EQU      27H

POSN        EQU      07H

DPST_ADD_LO      EQU      00H

DPST_ADD_HI      EQU      00H

SCRN_LO           EQU      80H        

SCRN_HI           EQU      02H        

ST_CHAR           EQU      20H

 

; RESPECTIVE CONSTANTS FOR GRAPHIC MODE

 

MODE_G            EQU      32H

PITCH_DT_G      EQU      77H

NUMCHR_G      EQU      1DH

DPST_ADD_LO_G     EQU      00H

DPST_ADD_HI_G     EQU      00H

SCRN_LO_G      EQU      00H        

SCRN_HI_G      EQU      0FH

ST_CHAR_G      EQU      01H

 

LAST_CHAR      EQU      29H         ; CODE FOR THE LAST CHAR. TO BE SENT

 

DISP_DATA      EQU      30H

CRSR_LO           EQU      31H

CRSR_HI           EQU      32H

 

MODE_SET      EQU      34H         ; TXT = 00H ; GPH = 0FFH

SCREEN1           EQU      35H

SCREEN            EQU      36H

 

CHAR_CODE      EQU      37H

LST_CHR           EQU      38H

 

;******************************************************************************

;CODE STARTS HERE

;******************************************************************************

 

            ORG      0000H

            JMP      POWER_ON

     

;******************************************************************************

; ALL THE PROCEDURES ARE WRITTEN HERE

;******************************************************************************

 

;******************   DELAY   ******************

;DELAY 1 SECOND

;***************************************************************

 

DLY1S:

        PUSH    B

        MOV     B,#04H

 

DREP1:  PUSH    B

        MOV     B,#0FFH

DREP:

 

        CALL    DLY1MS

 

        DJNZ    B,DREP

        POP     B

        DJNZ    B,DREP1

 

        POP     B

 

        RET

 

;*************************************************************

;DELAY FOR 1 msec

;****************************************************************

 

DLY1MS:         CALL    DELAY

                RET

 

;****************************************************************

;VARIABLE COUNT DELAY

;***************************************************************

 

_DELAY:

DELAY:

DE_LAY: PUSH      B

        MOV       B,#184

DLYLOOP:

        NOP

        NOP

        NOP

        DJNZ      B,DLYLOOP

      NOP

      NOP

      NOP

      NOP

     

        POP       B

      RET

 

;**************************   DELAY ENDS   **********************

 

INIT_DISP:

            CLR      P2.1        ; CHIP SELECT

 

            NOP

            NOP

           

            SETB      P3.1        ; SETTING RESET PIN HIGH FOR A PULSE

            LCALL      DLY1MS

 

            CLR      P3.1        ; GIVING RESET SIGNAL TO THE DISPLAY

                              ; THRU PORT PIN 3.1 (ACTIVE LOW)

 

            LCALL      DLY1MS

            LCALL      DLY1MS

            LCALL      DLY1MS

           

            SETB      P3.1        ; RESET PIN TAKEN HIGH

           

            SETB      P3.6        ; HIGH SIGNAL FOR GPH. MOD. READ

 

            NOP

            NOP

            NOP

            NOP

 

            SETB      P2.1        ; DISABLE CHIP

            RET

 

;*************         

 

BUSY_CHK:

            LCALL      DLY1MS

            LCALL      DLY1MS

           

            MOV      P0,#0FFH

            LCALL      DLY1MS

           

            SETB      P2.0        ; SETTING RS = 1 FOR INSTRUCTION

 

            NOP

            NOP

            NOP

            NOP

 

            CLR      P3.7        ; FOR ENABLE SIGNAL

            SETB      P3.6        ; RD SIGNAL FOR GRAPHIC MODULE R/W!

           

            NOP

            NOP

            NOP

            NOP

 

            CLR      P2.1        ; CHIP SELECT

 

            NOP

            NOP

            NOP

            NOP

           

BUSY:       MOV      A,P0        ; GET STATUS

            JB      P0.7,BUSY

 

            SETB      P2.1        ; DISABLE CHIP SELECT FOR GPH. MOD.

            SETB      P3.7        ; MAKING ENABLE SIG. INACTIVE

           

            RET

 

;*************

 

INST_WR:

            LCALL      DLY1MS

           

            MOV      P0,A

 

            SETB      P2.0        ; SETTING RS = 1 FOR INSTRUCTION

 

            NOP

            NOP

            NOP

           

            CLR      P3.6        ; WRITE SIGNAL FOR GPH. MOD. R/W! AS

                              ; WELL AS ENABLING ENABLE SIGNAL

            NOP

            NOP

            NOP

 

            CLR      P2.1        ; CHIP SELECT

 

            NOP

            NOP

            NOP

           

            SETB      P2.1        ; DISABLE CHIP

            SETB      P3.6        ; WRITE INACTIVE THRU R/W! = 1 AS WELL

                              ; AS DISABLING ENABLE SIGNAL

 

            LCALL      DLY1MS

 

            RET

           

;*************         

 

DATA_WR:

            LCALL      DLY1MS

           

            MOV      P0,A

 

            CLR      P2.0        ; SETTING RS = 0 FOR DATA

           

            NOP

            NOP

            NOP

 

            CLR      P3.6        ; WRITE SIGNAL FOR GPH. MOD. R/W! AS

                              ; WELL AS ENABLING ENABLE SIGNAL

            NOP

            NOP

            NOP

 

            CLR      P2.1        ; CHIP SELECT

           

            NOP

            NOP

            NOP

           

            SETB      P2.1        ; DISABLE CHIP

            SETB      P3.6        ; WRITE INACTIVE THRU R/W! = 1 AS WELL

                              ; AS DISABLING ENABLE SIGNAL

           

            LCALL      DLY1MS

           

            RET

 

 

;*************         

 

CHK_MODE:

            CLR      C           ; TEXT MODE IF CARRY SET

           

            MOV      A,MODE_SET

            CJNE      A,#00H,END_CHK

           

            SETB      C           ; GRAPHIC MODE IF CARRY RESET

 

END_CHK:      RET

 

;*************         

 

CHG_SCREEN:

            LCALL      CHK_MODE

 

            MOV      SCREEN1,#SCRN_HI_G      ; LOAD SCREEN SIZE FOR GPH.MODE

            MOV      SCREEN,#SCRN_LO_G

           

            JNC      CHANGE

           

            MOV      SCREEN1,#SCRN_HI      ; LOAD SCREEN SIZE FOR TXT.MODE

            MOV      SCREEN,#SCRN_LO

 

CHANGE:           RET

 

;*************

 

DP_MODE:                      ; THIS ROUTINE SETS THE MODE IN

            LCALL      BUSY_CHK    ; WHICH THE GPH.MODULE IS TO BE RUN

            MOV      A,#00H

            LCALL      INST_WR

            LCALL      CHK_MODE

     

            MOV      A,#MODE_G

            JNC      CHG_DPMOD   ; CARRY SET IN TEXT MODE

     

            MOV      A,#MODE

 

CHG_DPMOD:      LCALL      DATA_WR

 

            RET

           

;*************

 

CHR_PITCH:

            LCALL      BUSY_CHK    ; THE ROUTINE SETS THE CHAR.PITCH

            MOV      A,#01H

            LCALL      INST_WR

           

            LCALL      CHK_MODE

     

            MOV      A,#PITCH_DT_G

            JNC      CHG_PITCH   ; CARRY SET IN TEXT MODE

     

            MOV      A,#PITCH_DT

 

CHG_PITCH:      LCALL      DATA_WR

 

            RET

           

;*************

     

NUM_CHAR:

            LCALL      BUSY_CHK    ; NO.OF CHAR'S ARE SET

            MOV      A,#02H

            LCALL      INST_WR

 

            LCALL      CHK_MODE

     

            MOV      A,#NUMCHR_G

            JNC      CHG_NUMCHR  ; CARRY SET IN TEXT MODE

 

            MOV      A,#NUMCHR

           

CHG_NUMCHR:      LCALL      DATA_WR

           

            RET

           

;*************

 

TIME_DIV:

            LCALL      BUSY_CHK

            MOV      A,#03H

            LCALL      INST_WR

            MOV      A,#7FH

            LCALL      DATA_WR

            RET

           

;*************

 

CRSR_POS:

            LCALL      BUSY_CHK    ; CURSOR POSN ( VALID FOR TXT.MODE ) IS

            MOV      A,#04H            ; SET

            LCALL      INST_WR

            MOV      A,#POSN

            LCALL      DATA_WR

            RET

           

;*************

 

DP_ST_ADD:                    ; DISPLAY START ADDRESS IS SET

            LCALL      BUSY_CHK    ; TO START THE DISPLAY FROM THE DESIRED

                              ; LOCATION.

            MOV      A,#08H

            LCALL      INST_WR

           

            MOV      A,#DPST_ADD_LO

            LCALL      DATA_WR

           

            LCALL      BUSY_CHK

 

            MOV      A,#09H

            LCALL      INST_WR

           

            MOV      A,#DPST_ADD_HI

            LCALL      DATA_WR

     

            RET

           

;*************

 

CRSR_ADD:

            LCALL      BUSY_CHK    ; THIS SETS THE CURSOR FROM THE PT.

                              ; WHERE THE DISPLAY IS TO START

            MOV      A,#0AH

            LCALL      INST_WR

 

            MOV      A,CRSR_LO

            LCALL      DATA_WR

 

            LCALL      BUSY_CHK

 

            MOV      A,#0BH

            LCALL      INST_WR

 

            MOV      A,CRSR_HI

            LCALL      DATA_WR

 

            RET

           

;*************

 

WR_DPDATA:

            LCALL      BUSY_CHK    ; WRITES THE DATA TO THE SCREEN

            MOV      A,#0CH

            LCALL      INST_WR

            MOV      A,DISP_DATA

            LCALL      DATA_WR

            RET

           

;*************

 

CLR_DISP:                     ; CLEARS THE DISPLAY

           

            MOV      CRSR_HI,#00H

            MOV      CRSR_LO,#00H

           

            LCALL      CRSR_ADD    ; SET CURSOR ADDRESS

 

            LCALL      CHG_SCREEN

           

            MOV      R2,SCREEN1

            INC      R2

            MOV      R1,SCREEN

 

            LCALL      CHK_MODE

            JC      REP2

           

 

REP1:       MOV      DISP_DATA,#00H

            LCALL      WR_DPDATA   ; WRITE DISPLAY DATA

            DJNZ      R1,REP1    

            DJNZ      R2,REP1

 

            RET

           

REP2:       MOV      DISP_DATA,#20H

            LCALL      WR_DPDATA   ; WRITE DISPLAY DATA

            DJNZ      R1,REP2

            DJNZ      R2,REP2

 

            RET

           

;*************

 

MESSAGE:                      ; USED TO DISPLAY MESSAGES

            LCALL      CHK_MODE

            JNC      DO_NTHG

 

CONT_DSP:

            CLR      A

            MOVC      A,@A+DPTR

            CJNE      A,#10H,PROCEED

 

DO_NTHG:      RET

 

PROCEED:

            MOV      DISP_DATA,A

            LCALL      WR_DPDATA   ; WRITE DISPLAY DATA

 

            INC      DPTR

            JMP      CONT_DSP

 

            RET

 

;*************

 

;************************************************

; PROCEDURES END HERE

;************************************************

 

 

;******************************************************************************

                             

POWER_ON:  

            MOV      R1,#0FFH          ; POWER ON DELAY

P_DLY:            MOV      R7,#0FFH

P_DLY1:           DJNZ      R7,P_DLY1

            DJNZ      R1,P_DLY

 

            CLR      P1.0

 

            MOV      SP,#050H    ; INITIALIZING THE STACK POINTER

            MOV      IE,#00H           ; DISABLE ALL INTERRUPTS

            MOV      PSW,#00H    ; CLEAR THE PSW

           

            MOV      R5,#00H

            MOV      MODE_SET,#00H      ; SET TEXT MODE

           

CONTINUE:      LCALL      INIT_DISP

           

            LCALL      DLY1S

           

            LCALL      DP_MODE           ; SET DISPLAY MODE

 

            LCALL      DLY1S

 

            LCALL      CHR_PITCH   ; SET CHAR PITCH

            LCALL      NUM_CHAR    ; SET NO. OF CHARACTERS

            LCALL      TIME_DIV    ; SET NO. OF TIME DIVISIONS

            LCALL      CRSR_POS    ; SET CURSOR POSITION

            LCALL      DP_ST_ADD   ; SET DISPLAY START ADDRESS

           

            LCALL      CLR_DISP

 

            MOV      CRSR_HI,#00H

            MOV      CRSR_LO,#0F4H

           

            LCALL      CRSR_ADD    ; SET CURSOR ADDRESS

 

            MOV      DPTR,#CHAR_TAB

            LCALL      MESSAGE           ; MSG -> 'STARTING TEXT MODE...'

           

            LCALL      DLY1S

 

            LCALL      CHK_MODE

 

            MOV      R3,#ST_CHAR_G      ; LOAD THE STARTING CHAR.FOR THE

            JNC      NXT_CD            ; RESPECTIVE MODE OF DISPLAY

           

            MOV      R3,#ST_CHAR

           

NXT_CD:          

            MOV      CRSR_HI,#00H

            MOV      CRSR_LO,#00H

           

            LCALL      CRSR_ADD    ; SET CURSOR ADDRESS

 

            LCALL      CHG_SCREEN

 

            MOV      R2,SCREEN1

            INC      R2

            MOV      R1,SCREEN

 

            MOV      CHAR_CODE,R3

                       

REP:        MOV      DISP_DATA,CHAR_CODE

            LCALL      WR_DPDATA   ; WRITE DISPLAY DATA

            DJNZ      R1,REP           

            DJNZ      R2,REP

 

            INC      R3

            CJNE      R3,#LAST_CHAR,NXT_CD

           

            LCALL      CLR_DISP

 

            MOV      CRSR_HI,#00H

            MOV      CRSR_LO,#0F4H

           

            LCALL      CRSR_ADD    ; SET CURSOR ADDRESS

           

            LCALL      DLY1S

           

            MOV      DPTR,#CHAR_TAB1

            LCALL      MESSAGE           ; MSG -> 'WAIT...'

 

            MOV      CRSR_HI,#01H

            MOV      CRSR_LO,#42H

           

            LCALL      CRSR_ADD    ; SET CURSOR ADDRESS

           

            LCALL      DLY1S

 

            MOV      DPTR,#CHAR_TAB2

            LCALL      MESSAGE           ; MSG -> 'INITIALIZING GRAPHIC MODE...'

           

            LCALL      DLY1S      

            INC      R5

            MOV      MODE_SET,#0FFH

            CJNE      R5,#02H,CNT

           

            LCALL      INIT_DISP

           

            MOV      MODE_SET,#00H

 

            LCALL      CLR_DISP

 

            MOV      CRSR_HI,#00H

            MOV      CRSR_LO,#0F4H

           

            LCALL      CRSR_ADD    ; SET CURSOR ADDRESS

           

            LCALL      DLY1S

 

            MOV      DPTR,#CHAR_TAB3

            LCALL      MESSAGE           ; MSG -> 'TESTING OVER...'

                 

STOP:       SJMP      STOP

 

CNT:        LJMP      CONTINUE

 

;************************************************

 

CHAR_TAB:

            DB      'STARTING TEXT MODE...',10H

 

CHAR_TAB1:

            DB      'WAIT...',10H

 

CHAR_TAB2:

            DB      'INITIALIZING GRAPHIC MODE...',10H

 

CHAR_TAB3:

            DB      'TESTING OVER',10H

 

;************************************************

 

            END
